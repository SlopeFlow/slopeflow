ETH Open Reversion | Newsie v5

  • Aqua line = ETH open mid (18:00 ET)
  • Yellow lines = Prior session high/low
  • Yellow X = Prior session level swept (hard trigger)
  • Green triangle = LONG: bull candle closes above bear cluster high, AFTER prior session low swept
  • Red triangle = SHORT: bear candle closes below bull cluster low, AFTER prior session high swept
  • TP1 = top of bear cluster | TP2 = 50% above cluster top
  • Stop = 10pts below sweep low
//@version=6
indicator("ETH Open Reversion | Newsie v5", overlay=true, max_bars_back=1000)

// INPUTS
ext_min     = input.float(40.0, "Min Extension (pts)",        group="Setup")
ext_max     = input.float(80.0, "Max Extension (pts)",        group="Setup")
stop_pts    = input.float(10.0, "Stop Beyond Sweep Low (pts)", group="Setup")
max_cluster = input.int(3,      "Max cluster candles",        group="Setup")
line_len    = input.int(80,     "TP line length (bars)",      group="Display")

// SESSION & ETH OPEN
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")
is_session_open = et_hour == 18 and et_min == 0

var float eth_mid        = na
var float session_high   = na
var float session_low    = na
var float prev_sess_high = na
var float prev_sess_low  = na
var float day_high       = na
var float day_low        = na
var bool  setup_done     = false
var bool  swept_low_flag = false
var bool  swept_high_flag = false
var float sweep_low_val  = na
var float sweep_high_val = na

if is_session_open
    prev_sess_high  := session_high
    prev_sess_low   := session_low
    eth_mid         := (high + low) / 2
    session_high    := high
    session_low     := low
    day_high        := high
    day_low         := low
    setup_done      := false
    swept_low_flag  := false
    swept_high_flag := false
    sweep_low_val   := na
    sweep_high_val  := na

if not na(eth_mid)
    session_high := math.max(session_high, high)
    session_low  := math.min(session_low,  low)
    day_high     := math.max(day_high, high)
    day_low      := math.min(day_low,  low)

up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na
in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3

// Detect prior session level sweep (hard requirement: candle must touch/cross the level)
if not na(prev_sess_low) and low <= prev_sess_low and not swept_low_flag
    swept_low_flag := true
    sweep_low_val  := low

if not na(prev_sess_high) and high >= prev_sess_high and not swept_high_flag
    swept_high_flag := true
    sweep_high_val  := high

// CLUSTER DETECTION — up to max_cluster consecutive bear candles
bear1 = close[1] < open[1]
bear2 = close[2] < open[2]
bear3 = close[3] < open[3]
bull1 = close[1] > open[1]
bull2 = close[2] > open[2]
bull3 = close[3] > open[3]

// Cluster high = highest open of the bear candle sequence (top of cluster)
// Cluster low  = lowest low of the sequence (the sweep point)
cluster_bear_high = math.max(open[1], math.max(bear2 ? open[2] : open[1], bear3 ? open[3] : open[1]))
cluster_bear_low  = math.min(low[1],  math.min(bear2 ? low[2]  : low[1],  bear3 ? low[3]  : low[1]))
cluster_bear_range = cluster_bear_high - cluster_bear_low
tp1_long = cluster_bear_high  // TP1 = top of cluster (full cluster range reclaim)
tp2_long = cluster_bear_high + cluster_bear_range * 0.5  // TP2 = 50% above cluster top

cluster_bull_low  = math.min(open[1],  math.min(bull2 ? open[2]  : open[1],  bull3 ? open[3]  : open[1]))
cluster_bull_high = math.max(high[1],  math.max(bull2 ? high[2]  : high[1],  bull3 ? high[3]  : high[1]))
cluster_bull_range = cluster_bull_high - cluster_bull_low
tp1_short = cluster_bull_low
tp2_short = cluster_bull_low - cluster_bull_range * 0.5

// ENTRY CONDITIONS
// LONG: consecutive bear candles swept prior session low, current bull candle closes ABOVE cluster high
long_entry = close > open and                          // bull candle
             close > cluster_bear_high and             // closes above top of bear cluster
             bear1 and                                 // at least 1 bear candle before
             swept_low_flag and                        // prior session low was swept this session
             in_dn_zone and                            // in extension zone
             not setup_done

// SHORT: consecutive bull candles swept prior session high, current bear candle closes BELOW cluster low
short_entry = close < open and
              close < cluster_bull_low and
              bull1 and
              swept_high_flag and
              in_up_zone and
              not setup_done

if long_entry or short_entry
    setup_done := true

// PLOTS
plot(not na(eth_mid)        ? eth_mid        : na, "ETH Open Mid",    color=color.new(color.aqua,   30), linewidth=1)
plot(not na(prev_sess_high) ? prev_sess_high : na, "Prior Sess High", color=color.new(color.yellow, 20), linewidth=1)
plot(not na(prev_sess_low)  ? prev_sess_low  : na, "Prior Sess Low",  color=color.new(color.yellow, 20), linewidth=1)

// Sweep markers
plotshape(not na(prev_sess_low)  and low  <= prev_sess_low  and not swept_low_flag[1],  "Swept Low",  shape.xcross, location.belowbar, color=color.new(color.yellow, 0), size=size.tiny)
plotshape(not na(prev_sess_high) and high >= prev_sess_high and not swept_high_flag[1], "Swept High", shape.xcross, location.abovebar, color=color.new(color.yellow, 0), size=size.tiny)

// Signal arrows
plotshape(long_entry,  "Long",  shape.triangleup,   location.belowbar, color=color.new(color.lime, 0), size=size.normal)
plotshape(short_entry, "Short", shape.triangledown,  location.abovebar, color=color.new(color.red,  0), size=size.normal)

// TP / STOP LINES
var line tp1_line = na
var line tp2_line = na
var line sl_line  = na
atr = ta.atr(14)

if long_entry
    sl = cluster_bear_low - stop_pts
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1_long, bar_index + line_len, tp1_long, color=color.new(color.lime, 30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2_long, bar_index + line_len, tp2_long, color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,       bar_index + line_len, sl,       color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, low - atr*0.5,
              "LONG  SL:" + str.tostring(math.round(sl,0)) + "  TP1:" + str.tostring(math.round(tp1_long,0)),
              style=label.style_label_up, color=color.new(color.lime,5), textcolor=color.black, size=size.tiny)

if short_entry
    sl = cluster_bull_high + stop_pts
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1_short, bar_index + line_len, tp1_short, color=color.new(color.lime, 30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2_short, bar_index + line_len, tp2_short, color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,        bar_index + line_len, sl,        color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, high + atr*0.5,
              "SHORT  SL:" + str.tostring(math.round(sl,0)) + "  TP1:" + str.tostring(math.round(tp1_short,0)),
              style=label.style_label_down, color=color.new(color.red,5), textcolor=color.white, size=size.tiny)

alertcondition(long_entry,  "ETH LONG",  "ETH Reversion: LONG signal")
alertcondition(short_entry, "ETH SHORT", "ETH Reversion: SHORT signal")
