ETH Open Reversion | Newsie

Backtest: 83.3% win rate on TP1 | 62.5% on TP2 (24 setups, 2 months MNQ data)

  • Anchor: 18:00 ET 5m candle midpoint (ETH open)
  • Alert zone: 40-80pt extension from ETH mid (yellow bands)
  • Entry: 5m wick 5pt+ at extreme, next candle closes beyond wick
  • Stop: 10pts beyond wick extreme
  • TP1: 50% back to ETH mid (83.3% hit rate)
  • TP2: 75% back to ETH mid (runners, 62.5% hit rate)
//@version=6
indicator("ETH Open Reversion | Newsie", overlay=true, max_bars_back=1000)

// ─── INPUTS ───────────────────────────────────────────
ext_min     = input.float(40.0, "Min Extension (pts)",  group="Setup")
ext_max     = input.float(80.0, "Max Extension (pts)",  group="Setup")
wick_min    = input.float(5.0,  "Min Wick Size (pts)",  group="Setup")
stop_pts    = input.float(10.0, "Stop Beyond Wick (pts)", group="Setup")

show_eth    = input.bool(true,  "Show ETH Open Mid",    group="Display")
show_bands  = input.bool(true,  "Show STD Bands",       group="Display")
show_tp     = input.bool(true,  "Show TP Levels",       group="Display")

// ─── ETH OPEN CANDLE (18:00 ET) ───────────────────────
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")
et_mins = et_hour * 60 + et_min

is_eth_open = et_hour == 18 and et_min == 0
new_day     = ta.change(dayofmonth(time, "America/New_York")) != 0

var float eth_mid   = na
var float eth_high  = na
var float eth_low   = na
var float day_high  = na
var float day_low   = na
var bool  setup_done = false

// Reset on new day
if new_day
    day_high   := na
    day_low    := na
    setup_done := false

// Capture ETH open candle
if is_eth_open
    eth_mid    := (high + low) / 2
    eth_high   := high
    eth_low    := low
    day_high   := high
    day_low    := low
    setup_done := false

// Track running high/low from ETH open
if not na(eth_mid) and not is_eth_open
    day_high := math.max(day_high, high)
    day_low  := math.min(day_low, low)

// Extension from ETH mid
up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na

// ─── ETH MID LINE ─────────────────────────────────────
plot(show_eth and not na(eth_mid) ? eth_mid : na, "ETH Open Mid",
     color=color.new(color.white, 20), linewidth=2, style=plot.style_stepline)

// ─── STD EXTENSION BANDS ──────────────────────────────
plot(show_bands and not na(eth_mid) ? eth_mid + ext_min : na, "Min Ext Up",
     color=color.new(color.yellow, 60), linewidth=1, style=plot.style_stepline)
plot(show_bands and not na(eth_mid) ? eth_mid + ext_max : na, "Max Ext Up",
     color=color.new(color.yellow, 20), linewidth=1, style=plot.style_stepline)
plot(show_bands and not na(eth_mid) ? eth_mid - ext_min : na, "Min Ext Dn",
     color=color.new(color.yellow, 60), linewidth=1, style=plot.style_stepline)
plot(show_bands and not na(eth_mid) ? eth_mid - ext_max : na, "Max Ext Dn",
     color=color.new(color.yellow, 20), linewidth=1, style=plot.style_stepline)

// ─── SETUP DETECTION ──────────────────────────────────
in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3

// Change of Character (CoC) — 5m candle pattern
// SHORT CoC: prev candle has top wick >= wick_min, current closes below prev low
prev_top_wick  = high[1] - math.max(open[1], close[1])
prev_bot_wick  = math.min(open[1], close[1]) - low[1]

short_coc = prev_top_wick >= wick_min and close < low[1]
long_coc  = prev_bot_wick >= wick_min and close > high[1]

// Final signals
short_signal = in_up_zone and short_coc and not setup_done
long_signal  = in_dn_zone and long_coc  and not setup_done

// Mark setup as done once triggered
if short_signal or long_signal
    setup_done := true

// ─── TP / STOP LEVELS ─────────────────────────────────
var float entry_price = na
var float tp1_price   = na
var float tp2_price   = na
var float stop_price  = na

if long_signal
    entry_price := close
    stop_price  := day_low - stop_pts
    tp1_price   := eth_mid + (close - eth_mid) * 0.0  // = eth_mid is TP direction
    tp1_price   := close + (eth_mid - close) * 0.5    // 50% of way to eth_mid
    tp2_price   := close + (eth_mid - close) * 0.75   // 75% of way to eth_mid

if short_signal
    entry_price := close
    stop_price  := day_high + stop_pts
    tp1_price   := close - (close - eth_mid) * 0.5
    tp2_price   := close - (close - eth_mid) * 0.75

plot(show_tp and not na(tp1_price)  ? tp1_price  : na, "TP1 (50%)", color=color.new(color.lime, 20),   linewidth=1, style=plot.style_stepline)
plot(show_tp and not na(tp2_price)  ? tp2_price  : na, "TP2 (75%)", color=color.new(color.lime, 0),    linewidth=2, style=plot.style_stepline)
plot(show_tp and not na(stop_price) ? stop_price : na, "Stop",      color=color.new(color.red,  20),   linewidth=1, style=plot.style_stepline)

// ─── SIGNAL LABELS ────────────────────────────────────
atr = ta.atr(14)

if long_signal
    label.new(bar_index, low - atr * 0.3,
              "ETH LONG\nExt: " + str.tostring(math.round(dn_ext, 1)) + "pts\nWick: " + str.tostring(math.round(prev_bot_wick, 1)) + "pts",
              style=label.style_label_up, color=color.new(color.lime, 10),
              textcolor=color.black, size=size.small)

if short_signal
    label.new(bar_index, high + atr * 0.3,
              "ETH SHORT\nExt: " + str.tostring(math.round(up_ext, 1)) + "pts\nWick: " + str.tostring(math.round(prev_top_wick, 1)) + "pts",
              style=label.style_label_down, color=color.new(color.red, 10),
              textcolor=color.white, size=size.small)

// ─── BACKGROUND: IN ZONE ──────────────────────────────
bgcolor(in_up_zone ? color.new(color.red, 92) : in_dn_zone ? color.new(color.lime, 92) : na,
        title="Extension Zone")

// ─── ALERTS ───────────────────────────────────────────
alertcondition(long_signal,  "ETH Reversion LONG",  "ETH Open Reversion: LONG signal fired")
alertcondition(short_signal, "ETH Reversion SHORT", "ETH Open Reversion: SHORT signal fired")
