ETH Open Reversion | Newsie v6

  • Aqua = ETH open mid (18:00 ET anchor)
  • Yellow lines = Prior session high/low
  • Yellow X = Prior session level swept (hard requirement)
  • Green triangle = LONG signal (retest entry at 50% of confirmation candle)
  • Red triangle = SHORT signal
  • TP1 = 50% from sweep extreme back to ETH mid
  • TP2 = ETH open mid (runners)
  • Stop = 10pts beyond cluster extreme
  • Valid low = lowest cluster candle must be BEAR | Valid high = highest must be BULL
//@version=6
indicator("ETH Open Reversion | Newsie v6", overlay=true, max_bars_back=1000)

// INPUTS
ext_min     = input.float(40.0, "Min Extension (pts)",         group="Setup")
ext_max     = input.float(80.0, "Max Extension (pts)",         group="Setup")
stop_pts    = input.float(10.0, "Stop Beyond Sweep (pts)",     group="Setup")
line_len    = input.int(80,     "TP line length (bars)",       group="Display")

// SESSION & ETH OPEN (18:00 ET)
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")
is_session_open = et_hour == 18 and et_min == 0

var float eth_mid         = na
var float session_high    = na
var float session_low     = na
var float prev_sess_high  = na
var float prev_sess_low   = na
var float day_high        = na
var float day_low         = na
var bool  setup_done      = false
var bool  swept_low_flag  = false
var bool  swept_high_flag = false

if is_session_open
    prev_sess_high  := session_high
    prev_sess_low   := session_low
    eth_mid         := (high + low) / 2
    session_high    := high
    session_low     := low
    day_high        := high
    day_low         := low
    setup_done      := false
    swept_low_flag  := false
    swept_high_flag := false

if not na(eth_mid)
    session_high := math.max(session_high, high)
    session_low  := math.min(session_low,  low)
    day_high     := math.max(day_high, high)
    day_low      := math.min(day_low,  low)

// Extension
up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na
in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3

// Hard sweep detection
if not na(prev_sess_low)  and low  <= prev_sess_low  and not swept_low_flag
    swept_low_flag  := true
if not na(prev_sess_high) and high >= prev_sess_high and not swept_high_flag
    swept_high_flag := true

// Candle direction helpers
bear(i) => close[i] < open[i]
bull(i) => close[i] > open[i]

// Cluster detection
// LONG cluster: 1-3 consecutive bear candles, lowest bar must be BEAR
bear_cluster_valid = bear(1) and
    (bear(1) ? low[1] : na) <= (bear(2) ? low[2] : low[1]) ? bear(1) : bear(2)

// Cluster highs/lows (up to 3 bars back)
c_high_long = bear(2) ? math.max(high[1], high[2]) : high[1]
c_low_long  = bear(2) ? math.min(low[1],  low[2])  : low[1]

// Valid low: the bar with the lowest low must be bear
lowest_is_b1 = low[1] <= low[2]
valid_low_candle = lowest_is_b1 ? bear(1) : bear(2)

// Valid high (for short): bar with highest high must be bull
highest_is_b1 = high[1] >= high[2]
c_high_short = bull(2) ? math.max(high[1], high[2]) : high[1]
c_low_short  = bull(2) ? math.min(low[1],  low[2])  : low[1]
valid_high_candle = highest_is_b1 ? bull(1) : bull(2)

// Confirmation candles
bull_conf  = close > open and close > c_high_long
bear_conf  = close < open and close < c_low_short

// Entry signals
long_signal  = in_dn_zone and swept_low_flag  and bear(1) and bull_conf  and valid_low_candle  and not setup_done
short_signal = in_up_zone and swept_high_flag and bull(1) and bear_conf  and valid_high_candle and not setup_done

if long_signal or short_signal
    setup_done := true

// TP1 = 50% from sweep extreme back to ETH mid
tp1_long  = not na(eth_mid) ? c_low_long  + (eth_mid - c_low_long)  * 0.5 : na
tp1_short = not na(eth_mid) ? c_high_short - (c_high_short - eth_mid) * 0.5 : na
tp2_long  = eth_mid
tp2_short = eth_mid

// Plots
plot(not na(eth_mid)        ? eth_mid        : na, "ETH Open Mid",    color=color.new(color.aqua,   30), linewidth=1)
plot(not na(prev_sess_high) ? prev_sess_high : na, "Prior Sess High", color=color.new(color.yellow, 20), linewidth=1)
plot(not na(prev_sess_low)  ? prev_sess_low  : na, "Prior Sess Low",  color=color.new(color.yellow, 20), linewidth=1)

// Sweep markers
plotshape(not na(prev_sess_low)  and low  <= prev_sess_low  and not swept_low_flag[1],
          "Swept Low",  shape.xcross, location.belowbar, color=color.new(color.yellow,0), size=size.tiny)
plotshape(not na(prev_sess_high) and high >= prev_sess_high and not swept_high_flag[1],
          "Swept High", shape.xcross, location.abovebar, color=color.new(color.yellow,0), size=size.tiny)

// Signal arrows
plotshape(long_signal,  "Long",  shape.triangleup,   location.belowbar, color=color.new(color.lime,0), size=size.normal)
plotshape(short_signal, "Short", shape.triangledown,  location.abovebar, color=color.new(color.red, 0), size=size.normal)

// TP / Stop lines
var line tp1_line = na
var line tp2_line = na
var line sl_line  = na
atr = ta.atr(14)

if long_signal
    sl = c_low_long - stop_pts
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1_long, bar_index+line_len, tp1_long, color=color.new(color.lime,30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2_long, bar_index+line_len, tp2_long, color=color.new(color.lime,0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,       bar_index+line_len, sl,       color=color.new(color.red, 20), width=1, style=line.style_dashed)
    label.new(bar_index, low - atr*0.5,
              "LONG  SL:" + str.tostring(math.round(sl,0)) + "  TP1:" + str.tostring(math.round(tp1_long,0)),
              style=label.style_label_up, color=color.new(color.lime,5), textcolor=color.black, size=size.tiny)

if short_signal
    sl = c_high_short + stop_pts
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1_short, bar_index+line_len, tp1_short, color=color.new(color.lime,30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2_short, bar_index+line_len, tp2_short, color=color.new(color.lime,0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,        bar_index+line_len, sl,        color=color.new(color.red, 20), width=1, style=line.style_dashed)
    label.new(bar_index, high + atr*0.5,
              "SHORT  SL:" + str.tostring(math.round(sl,0)) + "  TP1:" + str.tostring(math.round(tp1_short,0)),
              style=label.style_label_down, color=color.new(color.red,5), textcolor=color.white, size=size.tiny)

alertcondition(long_signal,  "ETH LONG",  "ETH Reversion: LONG")
alertcondition(short_signal, "ETH SHORT", "ETH Reversion: SHORT")
