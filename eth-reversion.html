ETH Open Reversion | Newsie v4

  • Aqua line = ETH open mid (18:00 ET)
  • Yellow lines = Prior session high/low (liquidity targets)
  • Green triangle = LONG: bull candle closes above bear cluster high after sweeping prior session low
  • Red triangle = SHORT: bear candle closes below bull cluster low after sweeping prior session high
  • TP1 = 50% of cluster range | TP2 = 50% back to ETH mid (runners)
  • Stop = 10pts beyond cluster extreme
//@version=6
indicator("ETH Open Reversion | Newsie v4", overlay=true, max_bars_back=1000)

// INPUTS
ext_min      = input.float(40.0, "Min Extension (pts)",       group="Setup")
ext_max      = input.float(80.0, "Max Extension (pts)",       group="Setup")
stop_pts     = input.float(10.0, "Stop Beyond Cluster (pts)", group="Setup")
sweep_range  = input.float(15.0, "Prior Level Sweep Range (pts)", group="Setup")
line_len     = input.int(60,     "TP line length (bars)",     group="Display")

// PRIOR SESSION HIGH/LOW
// Session = 18:00 ET to 17:00 ET next day
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")
et_mins = et_hour * 60 + et_min

is_session_open = et_hour == 18 and et_min == 0
new_day = ta.change(dayofmonth(time, "America/New_York")) != 0

var float eth_mid       = na
var float session_high  = na
var float session_low   = na
var float prev_sess_high = na
var float prev_sess_low  = na
var float day_high      = na
var float day_low       = na
var bool  setup_done    = false

// On new session open: roll current session to prior, reset current
if is_session_open
    prev_sess_high := session_high
    prev_sess_low  := session_low
    eth_mid        := (high + low) / 2
    session_high   := high
    session_low    := low
    day_high       := high
    day_low        := low
    setup_done     := false

// Track current session high/low
if not na(eth_mid)
    session_high := math.max(session_high, high)
    session_low  := math.min(session_low,  low)
    day_high     := math.max(day_high, high)
    day_low      := math.min(day_low,  low)

up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na

in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3

// ETH MID LINE
plot(not na(eth_mid) ? eth_mid : na, "ETH Open Mid",
     color=color.new(color.aqua, 30), linewidth=1, style=plot.style_line)

// PRIOR SESSION LEVELS
plot(not na(prev_sess_high) ? prev_sess_high : na, "Prior Sess High",
     color=color.new(color.yellow, 20), linewidth=1, style=plot.style_line)
plot(not na(prev_sess_low) ? prev_sess_low : na, "Prior Sess Low",
     color=color.new(color.yellow, 20), linewidth=1, style=plot.style_line)

// SWEEP DETECTION
// Price wicked through prior level but close is back inside
near_prior_high = not na(prev_sess_high) and math.abs(high - prev_sess_high) <= sweep_range
near_prior_low  = not na(prev_sess_low)  and math.abs(low  - prev_sess_low)  <= sweep_range

swept_high = near_prior_high and high >= prev_sess_high and close < prev_sess_high
swept_low  = near_prior_low  and low  <= prev_sess_low  and close > prev_sess_low

// CLUSTER LOW/HIGH DETECTION
// Find the lowest low of the current bearish cluster (up to 3 consecutive bear candles)
bear_candle  = close < open
bull_candle  = close > open

// For LONG: identify cluster of 1-3 bear candles at the low
// Entry: bull candle closes above the HIGH of the cluster (first bear candle in sequence)
cluster_start_high = math.max(high[1], math.max(high[2], high[3]))
cluster_low        = math.min(low[1],  math.min(low[2],  low[3]))
cluster_range      = cluster_start_high - cluster_low
tp1_long           = cluster_low + cluster_range * 0.5  // 50% of cluster range

// Long entry: bull close above cluster high, after sweep of prior low, in extension zone
long_entry = bull_candle and close > cluster_start_high and
             (swept_low or (not na(prev_sess_low) and day_low <= prev_sess_low + sweep_range)) and
             in_dn_zone and not setup_done

// For SHORT: cluster of 1-3 bull candles at the high
cluster_start_low  = math.min(low[1],  math.min(low[2],  low[3]))
cluster_high_val   = math.max(high[1], math.max(high[2], high[3]))
cluster_range_s    = cluster_high_val - cluster_start_low
tp1_short          = cluster_high_val - cluster_range_s * 0.5

short_entry = bear_candle and close < cluster_start_low and
              (swept_high or (not na(prev_sess_high) and day_high >= prev_sess_high - sweep_range)) and
              in_up_zone and not setup_done

if long_entry or short_entry
    setup_done := true

// ARROWS
plotshape(long_entry,  "Long",  shape.triangleup,   location.belowbar, color=color.new(color.lime, 0), size=size.normal)
plotshape(short_entry, "Short", shape.triangledown,  location.abovebar, color=color.new(color.red,  0), size=size.normal)

// TP / STOP LINES
var line tp1_line = na
var line tp2_line = na
var line sl_line  = na
atr = ta.atr(14)

if long_entry
    sl  = cluster_low - stop_pts
    tp2 = close + (eth_mid - close) * 0.5
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1_long,  bar_index + line_len, tp1_long,  color=color.new(color.lime, 20), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2,        bar_index + line_len, tp2,       color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,         bar_index + line_len, sl,        color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, low - atr*0.4,
              "LONG  SL:" + str.tostring(math.round(sl,0)) + "  TP1:" + str.tostring(math.round(tp1_long,0)),
              style=label.style_label_up, color=color.new(color.lime,5), textcolor=color.black, size=size.tiny)

if short_entry
    sl  = cluster_high_val + stop_pts
    tp2 = close - (close - eth_mid) * 0.5
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1_short, bar_index + line_len, tp1_short, color=color.new(color.lime, 20), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2,       bar_index + line_len, tp2,       color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,        bar_index + line_len, sl,        color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, high + atr*0.4,
              "SHORT  SL:" + str.tostring(math.round(sl,0)) + "  TP1:" + str.tostring(math.round(tp1_short,0)),
              style=label.style_label_down, color=color.new(color.red,5), textcolor=color.white, size=size.tiny)

alertcondition(long_entry,  "ETH LONG",  "ETH Reversion: LONG signal")
alertcondition(short_entry, "ETH SHORT", "ETH Reversion: SHORT signal")
