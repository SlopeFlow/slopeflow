ETH Open Reversion | Newsie v3

Clean minimal design:

  • Single gray dashed line = ETH open mid (18:00 ET)
  • Small dot = price enters extension zone (40-80pts from mid)
  • Green triangle up = LONG signal | Red triangle down = SHORT signal
  • Dashed lines = TP1, TP2, Stop — auto-expire after 40 bars
  • No background shading, no spanning zone lines
//@version=6
indicator("ETH Open Reversion | Newsie", overlay=true, max_bars_back=1000)

// INPUTS
ext_min  = input.float(40.0, "Min Extension (pts)",    group="Setup")
ext_max  = input.float(80.0, "Max Extension (pts)",    group="Setup")
wick_min = input.float(5.0,  "Min Wick Size (pts)",    group="Setup")
stop_pts = input.float(10.0, "Stop Beyond Wick (pts)", group="Setup")
tp1_pts  = input.float(20.0, "TP1 (pts)",              group="Setup")
tp2_pts  = input.float(40.0, "TP2 (pts)",              group="Setup")
line_len = input.int(40,     "TP/Stop line length (bars)", group="Display")

// ETH OPEN
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")
is_eth_open = et_hour == 18 and et_min == 0
new_day     = ta.change(dayofmonth(time, "America/New_York")) != 0

var float eth_mid  = na
var float day_high = na
var float day_low  = na
var bool  setup_done = false

if new_day
    day_high := na
    day_low  := na
    setup_done := false

if is_eth_open
    eth_mid    := (high + low) / 2
    day_high   := high
    day_low    := low
    setup_done := false

if not na(eth_mid) and not is_eth_open
    day_high := math.max(day_high, high)
    day_low  := math.min(day_low,  low)

up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na

in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3

// ETH MID — single dashed line, subtle
plot(not na(eth_mid) ? eth_mid : na, "ETH Open Mid",
     color=color.new(color.gray, 50), linewidth=1, style=plot.style_line)

// CoC signals
prev_top_wick = high[1] - math.max(open[1], close[1])
prev_bot_wick = math.min(open[1], close[1]) - low[1]
short_coc = prev_top_wick >= wick_min and close < low[1]
long_coc  = prev_bot_wick >= wick_min and close > high[1]

long_signal  = in_dn_zone and long_coc  and not setup_done
short_signal = in_up_zone and short_coc and not setup_done

if long_signal or short_signal
    setup_done := true

// ARROWS only — no labels
plotshape(long_signal,  "Long",  shape.triangleup,   location.belowbar, color=color.new(color.lime, 0), size=size.normal)
plotshape(short_signal, "Short", shape.triangledown,  location.abovebar, color=color.new(color.red,  0), size=size.normal)

// TP / STOP lines — draw once, auto-expire after line_len bars
var line tp1_line = na
var line tp2_line = na
var line sl_line  = na

if long_signal
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, close + tp1_pts, bar_index + line_len, close + tp1_pts,
                         color=color.new(color.lime, 30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, close + tp2_pts, bar_index + line_len, close + tp2_pts,
                         color=color.new(color.lime, 10), width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, close - stop_pts, bar_index + line_len, close - stop_pts,
                         color=color.new(color.red, 30), width=1, style=line.style_dashed)

if short_signal
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, close - tp1_pts, bar_index + line_len, close - tp1_pts,
                         color=color.new(color.lime, 30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, close - tp2_pts, bar_index + line_len, close - tp2_pts,
                         color=color.new(color.lime, 10), width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, close + stop_pts, bar_index + line_len, close + stop_pts,
                         color=color.new(color.red, 30), width=1, style=line.style_dashed)

// Tiny zone dot at extremes only (no spanning lines)
plotshape(in_up_zone and not in_up_zone[1], "Enter Short Zone", shape.circle,
          location.abovebar, color=color.new(color.red, 40), size=size.tiny)
plotshape(in_dn_zone and not in_dn_zone[1], "Enter Long Zone", shape.circle,
          location.belowbar, color=color.new(color.lime, 40), size=size.tiny)

alertcondition(long_signal,  "ETH LONG",  "ETH Reversion: LONG signal")
alertcondition(short_signal, "ETH SHORT", "ETH Reversion: SHORT signal")
