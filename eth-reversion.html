ETH Open Reversion | Newsie v2

83.3% win rate TP1 | 62.5% TP2 — 24 setups, 2 months MNQ

  • White line = ETH open mid (18:00 ET anchor)
  • Yellow dashed bands = 40pt and 80pt extension zones
  • Green label = LONG signal | Red label = SHORT signal
  • Dashed lines = TP1, TP2, Stop (auto-clear on new signal)
//@version=6
indicator("ETH Open Reversion | Newsie", overlay=true, max_bars_back=1000)

// INPUTS
ext_min  = input.float(40.0, "Min Extension (pts)",    group="Setup")
ext_max  = input.float(80.0, "Max Extension (pts)",    group="Setup")
wick_min = input.float(5.0,  "Min Wick Size (pts)",    group="Setup")
stop_pts = input.float(10.0, "Stop Beyond Wick (pts)", group="Setup")

// ETH OPEN CANDLE (18:00 ET)
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")

is_eth_open = et_hour == 18 and et_min == 0
new_day     = ta.change(dayofmonth(time, "America/New_York")) != 0

var float eth_mid   = na
var float day_high  = na
var float day_low   = na
var bool  setup_done = false

if new_day
    day_high   := na
    day_low    := na
    setup_done := false

if is_eth_open
    eth_mid    := (high + low) / 2
    day_high   := high
    day_low    := low
    setup_done := false

if not na(eth_mid) and not is_eth_open
    day_high := math.max(day_high, high)
    day_low  := math.min(day_low,  low)

up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na

// ETH MID LINE
plot(not na(eth_mid) ? eth_mid : na, "ETH Open Mid",
     color=color.new(color.white, 30), linewidth=2, style=plot.style_stepline)

// EXTENSION BANDS (subtle)
plot(not na(eth_mid) ? eth_mid + ext_min : na, "40pt Up",  color=color.new(color.yellow, 75), linewidth=1, style=plot.style_stepline)
plot(not na(eth_mid) ? eth_mid + ext_max : na, "80pt Up",  color=color.new(color.yellow, 50), linewidth=1, style=plot.style_stepline)
plot(not na(eth_mid) ? eth_mid - ext_min : na, "40pt Dn",  color=color.new(color.yellow, 75), linewidth=1, style=plot.style_stepline)
plot(not na(eth_mid) ? eth_mid - ext_max : na, "80pt Dn",  color=color.new(color.yellow, 50), linewidth=1, style=plot.style_stepline)

// ZONE HIGHLIGHT (very subtle)
in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3
bgcolor(in_up_zone ? color.new(color.red, 96) : in_dn_zone ? color.new(color.lime, 96) : na)

// CHANGE OF CHARACTER
prev_top_wick = high[1] - math.max(open[1], close[1])
prev_bot_wick = math.min(open[1], close[1]) - low[1]
short_coc = prev_top_wick >= wick_min and close < low[1]
long_coc  = prev_bot_wick >= wick_min and close > high[1]

long_signal  = in_dn_zone and long_coc  and not setup_done
short_signal = in_up_zone and short_coc and not setup_done

if long_signal or short_signal
    setup_done := true

// TP/STOP LINES — draw only on signal, auto-clear previous
var line tp1_line  = na
var line tp2_line  = na
var line sl_line   = na

atr = ta.atr(14)

if long_signal
    tp1 = close + (eth_mid - close) * 0.5
    tp2 = close + (eth_mid - close) * 0.75
    sl  = day_low - stop_pts
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1, bar_index + 30, tp1, color=color.new(color.lime, 20),  width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2, bar_index + 30, tp2, color=color.new(color.lime, 0),   width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,  bar_index + 30, sl,  color=color.new(color.red,  20),  width=1, style=line.style_dashed)
    label.new(bar_index, low - atr * 0.4,
              "ETH LONG  " + str.tostring(math.round(dn_ext, 0)) + "pt | W:" + str.tostring(math.round(prev_bot_wick, 1)),
              style=label.style_label_up, color=color.new(color.lime, 5), textcolor=color.black, size=size.small)

if short_signal
    tp1 = close - (close - eth_mid) * 0.5
    tp2 = close - (close - eth_mid) * 0.75
    sl  = day_high + stop_pts
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, tp1, bar_index + 30, tp1, color=color.new(color.lime, 20),  width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, tp2, bar_index + 30, tp2, color=color.new(color.lime, 0),   width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, sl,  bar_index + 30, sl,  color=color.new(color.red,  20),  width=1, style=line.style_dashed)
    label.new(bar_index, high + atr * 0.4,
              "ETH SHORT  " + str.tostring(math.round(up_ext, 0)) + "pt | W:" + str.tostring(math.round(prev_top_wick, 1)),
              style=label.style_label_down, color=color.new(color.red, 5), textcolor=color.white, size=size.small)

alertcondition(long_signal,  "ETH Reversion LONG",  "ETH Open Reversion: LONG")
alertcondition(short_signal, "ETH Reversion SHORT", "ETH Open Reversion: SHORT")
