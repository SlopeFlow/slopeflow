23/29/35 Time Reversal | Newsie

  • Purple bar = qualifying time bar (hour+min OR minute = 22-24, 28-30, 34-36)
  • Number label = which zone (23, 29, or 35)
  • Green triangle = bull candle at time bar (LONG signal)
  • Red triangle = bear candle at time bar (SHORT signal)
  • Dashed green = TP at 2:1 R:R | Dashed red = Stop 10pts
  • Load on 1m chart | Works any time of day
//@version=6
indicator("23/29/35 Time Reversal | Newsie", overlay=true, max_bars_back=500)

// INPUTS
stop_pts   = input.float(10.0, "Stop Loss (pts)",   group="Risk")
rr         = input.float(2.0,  "R:R Target",        group="Risk")
show_zones = input.bool(true,  "Highlight time bars", group="Display")

// TIME CALCULATION (ET)
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")

h_plus_m = et_hour + et_min
min_only = et_min

// Qualifying time check
in_23 = (h_plus_m >= 22 and h_plus_m <= 24) or (min_only >= 22 and min_only <= 24)
in_29 = (h_plus_m >= 28 and h_plus_m <= 30) or (min_only >= 28 and min_only <= 30)
in_35 = (h_plus_m >= 34 and h_plus_m <= 36) or (min_only >= 34 and min_only <= 36)

is_time_bar = in_23 or in_29 or in_35

// Time zone label
time_zone = in_23 ? "23" : in_29 ? "29" : in_35 ? "35" : ""

// Reversal candle detection
// Bear reversal: bear candle at a time bar (potential short)
// Bull reversal: bull candle at a time bar (potential long)
bear_rev = is_time_bar and close < open
bull_rev = is_time_bar and close > open

// TP = stop * R:R
tp_pts = stop_pts * rr

// Highlight qualifying time bars (subtle background)
bgcolor(is_time_bar ? color.new(color.purple, 90) : na, title="Time Bar Zone")

// Mark the reversal candles
plotshape(bull_rev, "Bull Reversal", shape.triangleup,   location.belowbar,
          color=color.new(color.lime, 0), size=size.small)
plotshape(bear_rev, "Bear Reversal", shape.triangledown,  location.abovebar,
          color=color.new(color.red,  0), size=size.small)

// Labels with time zone number
if bull_rev
    label.new(bar_index, low - ta.atr(14)*0.4,
              time_zone,
              style=label.style_label_up,
              color=color.new(color.lime, 10),
              textcolor=color.black,
              size=size.tiny)

if bear_rev
    label.new(bar_index, high + ta.atr(14)*0.4,
              time_zone,
              style=label.style_label_down,
              color=color.new(color.red, 10),
              textcolor=color.white,
              size=size.tiny)

// TP / Stop lines on reversal candles
var line tp_line = na
var line sl_line = na

if bull_rev
    line.delete(tp_line)
    line.delete(sl_line)
    tp_line := line.new(bar_index, close + tp_pts,  bar_index + 30, close + tp_pts,
                        color=color.new(color.lime, 20), width=1, style=line.style_dashed)
    sl_line := line.new(bar_index, close - stop_pts, bar_index + 30, close - stop_pts,
                        color=color.new(color.red,  20), width=1, style=line.style_dashed)

if bear_rev
    line.delete(tp_line)
    line.delete(sl_line)
    tp_line := line.new(bar_index, close - tp_pts,  bar_index + 30, close - tp_pts,
                        color=color.new(color.lime, 20), width=1, style=line.style_dashed)
    sl_line := line.new(bar_index, close + stop_pts, bar_index + 30, close + stop_pts,
                        color=color.new(color.red,  20), width=1, style=line.style_dashed)

alertcondition(bull_rev, "23/29/35 LONG",  "Time Reversal: LONG signal")
alertcondition(bear_rev, "23/29/35 SHORT", "Time Reversal: SHORT signal")
alertcondition(in_23 and is_time_bar, "Zone 23", "Time bar: 23 zone")
alertcondition(in_29 and is_time_bar, "Zone 29", "Time bar: 29 zone")
alertcondition(in_35 and is_time_bar, "Zone 35", "Time bar: 35 zone")
