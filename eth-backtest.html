ETH Open Reversion | Backtest v5

Run on MNQ1! 5m chart — Strategy Tester tab.

  • Hard prior session level sweep required before any signal
  • Entry: bull/bear candle closes beyond 1-2 candle cluster after sweep
  • TP1 = top of cluster | Stop = 10pts beyond sweep extreme
//@version=6
strategy("ETH Open Reversion | Backtest v5", overlay=true, max_bars_back=1000,
         initial_capital=50000, default_qty_type=strategy.fixed, default_qty_value=1,
         commission_type=strategy.commission.cash_per_contract, commission_value=2.05,
         slippage=2)

// INPUTS
ext_min     = input.float(40.0, "Min Extension (pts)",         group="Setup")
ext_max     = input.float(80.0, "Max Extension (pts)",         group="Setup")
stop_pts    = input.float(10.0, "Stop Beyond Sweep Low (pts)", group="Setup")
line_len    = input.int(60,     "TP line length (bars)",       group="Display")

// SESSION & ETH OPEN
et_hour = hour(time, "America/New_York")
et_min  = minute(time, "America/New_York")
is_session_open = et_hour == 18 and et_min == 0

var float eth_mid         = na
var float session_high    = na
var float session_low     = na
var float prev_sess_high  = na
var float prev_sess_low   = na
var float day_high        = na
var float day_low         = na
var bool  setup_done      = false
var bool  swept_low_flag  = false
var bool  swept_high_flag = false

if is_session_open
    prev_sess_high  := session_high
    prev_sess_low   := session_low
    eth_mid         := (high + low) / 2
    session_high    := high
    session_low     := low
    day_high        := high
    day_low         := low
    setup_done      := false
    swept_low_flag  := false
    swept_high_flag := false

if not na(eth_mid)
    session_high := math.max(session_high, high)
    session_low  := math.min(session_low,  low)
    day_high     := math.max(day_high, high)
    day_low      := math.min(day_low,  low)

up_ext = not na(day_high) and not na(eth_mid) ? day_high - eth_mid : na
dn_ext = not na(day_low)  and not na(eth_mid) ? eth_mid - day_low  : na
in_up_zone = not na(up_ext) and up_ext >= ext_min and up_ext <= ext_max and up_ext > dn_ext * 1.3
in_dn_zone = not na(dn_ext) and dn_ext >= ext_min and dn_ext <= ext_max and dn_ext > up_ext * 1.3

// Hard sweep detection
if not na(prev_sess_low)  and low  <= prev_sess_low  and not swept_low_flag
    swept_low_flag  := true
if not na(prev_sess_high) and high >= prev_sess_high and not swept_high_flag
    swept_high_flag := true

// Cluster detection (5m bars)
bear1 = close[1] < open[1]
bear2 = close[2] < open[2]
bull1 = close[1] > open[1]
bull2 = close[2] > open[2]

cluster_bear_high = math.max(open[1], bear2 ? open[2] : open[1])
cluster_bear_low  = math.min(low[1],  bear2 ? low[2]  : low[1])
cluster_bear_range = cluster_bear_high - cluster_bear_low

cluster_bull_low  = math.min(open[1],  bull2 ? open[2]  : open[1])
cluster_bull_high = math.max(high[1],  bull2 ? high[2]  : high[1])
cluster_bull_range = cluster_bull_high - cluster_bull_low

// Entry conditions
long_entry  = close > open and close > cluster_bear_high and bear1 and
              swept_low_flag and in_dn_zone and not setup_done and strategy.position_size == 0

short_entry = close < open and close < cluster_bull_low and bull1 and
              swept_high_flag and in_up_zone and not setup_done and strategy.position_size == 0

if long_entry or short_entry
    setup_done := true

// Display
plot(not na(eth_mid)        ? eth_mid        : na, "ETH Mid",         color=color.new(color.aqua,   30), linewidth=1)
plot(not na(prev_sess_high) ? prev_sess_high : na, "Prior Sess High", color=color.new(color.yellow, 20), linewidth=1)
plot(not na(prev_sess_low)  ? prev_sess_low  : na, "Prior Sess Low",  color=color.new(color.yellow, 20), linewidth=1)

plotshape(long_entry,  "Long",  shape.triangleup,   location.belowbar, color=color.new(color.lime, 0), size=size.normal)
plotshape(short_entry, "Short", shape.triangledown,  location.abovebar, color=color.new(color.red,  0), size=size.normal)

// Entries & exits
atr = ta.atr(14)

if long_entry
    tp1 = cluster_bear_high
    tp2 = cluster_bear_high + cluster_bear_range * 0.5
    sl  = cluster_bear_low - stop_pts
    strategy.entry("Long", strategy.long)
    strategy.exit("Long Exit", "Long", limit=tp1, stop=sl)

if short_entry
    tp1 = cluster_bull_low
    tp2 = cluster_bull_low - cluster_bull_range * 0.5
    sl  = cluster_bull_high + stop_pts
    strategy.entry("Short", strategy.short)
    strategy.exit("Short Exit", "Short", limit=tp1, stop=sl)

if et_hour == 21 and et_min == 55 and strategy.position_size != 0
    strategy.close_all("Session End")
