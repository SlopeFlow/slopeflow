CHoCH + 23/29/35 Time Reversal | Newsie

  • Green dotted line = Most recent valid CHoCH high (bull candle pivot)
  • Red dotted line = Most recent valid CHoCH low (bear candle pivot)
  • Yellow X = Liquidity sweep (wick beyond CHoCH level, close back inside)
  • Purple background = Qualifying time bar (23/29/35 zone)
  • Green triangle + number = LONG signal (swept low + time bar + bull candle)
  • Red triangle + number = SHORT signal (swept high + time bar + bear candle)
  • One signal per sweep — resets when new CHoCH forms
  • Stop: 10pts | TP: 20pts (2:1 R:R)
//@version=6
indicator("CHoCH + 23/29/35 | Newsie", overlay=true, max_bars_back=1000)

// INPUTS
stop_pts    = input.float(10.0, "Stop Loss (pts)",      group="Risk")
rr          = input.float(2.0,  "R:R Target",           group="Risk")
swing_len   = input.int(5,      "Swing lookback bars",  group="Structure")
sweep_range = input.float(5.0,  "Sweep tolerance (pts)", group="Structure")
line_len    = input.int(60,     "Line length (bars)",   group="Display")

// TIME ZONES (ET)
et_hour  = hour(time, "America/New_York")
et_min   = minute(time, "America/New_York")
h_plus_m = et_hour + et_min
min_only = et_min

in_23 = (h_plus_m >= 22 and h_plus_m <= 24) or (min_only >= 22 and min_only <= 24)
in_29 = (h_plus_m >= 28 and h_plus_m <= 30) or (min_only >= 28 and min_only <= 30)
in_35 = (h_plus_m >= 34 and h_plus_m <= 36) or (min_only >= 34 and min_only <= 36)
is_time_bar = in_23 or in_29 or in_35
time_zone   = in_23 ? "23" : in_29 ? "29" : "35"

// SWING HIGH/LOW DETECTION
// Valid swing high: highest high in lookback, and that bar must be a BULL candle
// Valid swing low:  lowest low in lookback, and that bar must be a BEAR candle
pivot_high_val = ta.pivothigh(high, swing_len, swing_len)
pivot_low_val  = ta.pivotlow(low,  swing_len, swing_len)

// Only accept bull candle highs and bear candle lows
bull_pivot_high = not na(pivot_high_val) and close[swing_len] > open[swing_len]
bear_pivot_low  = not na(pivot_low_val)  and close[swing_len] < open[swing_len]

// Track most recent valid CHoCH high and low
var float choch_high = na
var float choch_low  = na
var int   choch_high_bar = na
var int   choch_low_bar  = na

if bull_pivot_high
    choch_high     := pivot_high_val
    choch_high_bar := bar_index - swing_len

if bear_pivot_low
    choch_low     := pivot_low_val
    choch_low_bar := bar_index - swing_len

// Draw CHoCH levels as horizontal lines
var line high_line = na
var line low_line  = na

if bull_pivot_high
    line.delete(high_line)
    high_line := line.new(bar_index - swing_len, choch_high,
                          bar_index - swing_len + line_len, choch_high,
                          color=color.new(color.lime, 30), width=1, style=line.style_dotted)

if bear_pivot_low
    line.delete(low_line)
    low_line := line.new(bar_index - swing_len, choch_low,
                         bar_index - swing_len + line_len, choch_low,
                         color=color.new(color.red, 30), width=1, style=line.style_dotted)

// SWEEP DETECTION
// Sweep high: wick above CHoCH high but candle body closes back below it
swept_high = not na(choch_high) and high > choch_high and close < choch_high
swept_low  = not na(choch_low)  and low  < choch_low  and close > choch_low

// Track sweep flags (reset when new CHoCH forms)
var bool swept_high_flag = false
var bool swept_low_flag  = false

if bull_pivot_high
    swept_high_flag := false
if bear_pivot_low
    swept_low_flag := false

if swept_high
    swept_high_flag := true
if swept_low
    swept_low_flag := true

// Sweep markers
plotshape(swept_high and not swept_high[1], "Swept High",
          shape.xcross, location.abovebar, color=color.new(color.yellow, 0), size=size.tiny)
plotshape(swept_low and not swept_low[1], "Swept Low",
          shape.xcross, location.belowbar, color=color.new(color.yellow, 0), size=size.tiny)

// SIGNALS — time bar + swept level + reversal candle
bear_candle = close < open
bull_candle = close > open

// SHORT: CHoCH high was swept, time bar hits, bear candle = reversal
short_signal = swept_high_flag and is_time_bar and bear_candle

// LONG: CHoCH low was swept, time bar hits, bull candle = reversal
long_signal  = swept_low_flag  and is_time_bar and bull_candle

// One signal per sweep (reset after entry)
var bool long_taken  = false
var bool short_taken = false

if bear_pivot_low:  long_taken  := false
if bull_pivot_high: short_taken := false
if swept_low:       long_taken  := false
if swept_high:      short_taken := false

long_entry  = long_signal  and not long_taken
short_entry = short_signal and not short_taken

if long_entry:  long_taken  := true
if short_entry: short_taken := true

// Background highlight on time bars
bgcolor(is_time_bar ? color.new(color.purple, 92) : na, title="Time Bar")

// Arrows
plotshape(long_entry,  "Long",  shape.triangleup,   location.belowbar,
          color=color.new(color.lime, 0), size=size.normal)
plotshape(short_entry, "Short", shape.triangledown,  location.abovebar,
          color=color.new(color.red,  0), size=size.normal)

// TP / Stop lines
tp_pts = stop_pts * rr
var line tp_line = na
var line sl_line = na
atr = ta.atr(14)

if long_entry
    line.delete(tp_line)
    line.delete(sl_line)
    tp_line := line.new(bar_index, close + tp_pts,  bar_index + line_len, close + tp_pts,
                        color=color.new(color.lime, 20), width=1, style=line.style_dashed)
    sl_line := line.new(bar_index, close - stop_pts, bar_index + line_len, close - stop_pts,
                        color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, low - atr*0.4, time_zone,
              style=label.style_label_up, color=color.new(color.lime,5),
              textcolor=color.black, size=size.tiny)

if short_entry
    line.delete(tp_line)
    line.delete(sl_line)
    tp_line := line.new(bar_index, close - tp_pts,  bar_index + line_len, close - tp_pts,
                        color=color.new(color.lime, 20), width=1, style=line.style_dashed)
    sl_line := line.new(bar_index, close + stop_pts, bar_index + line_len, close + stop_pts,
                        color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, high + atr*0.4, time_zone,
              style=label.style_label_down, color=color.new(color.red,5),
              textcolor=color.white, size=size.tiny)

alertcondition(long_entry,  "CHoCH LONG",  "CHoCH + Time: LONG signal")
alertcondition(short_entry, "CHoCH SHORT", "CHoCH + Time: SHORT signal")
