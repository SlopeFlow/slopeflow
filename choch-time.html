23/29/35 CHoCH Reversion | Newsie

  • Purple bar = 23/29/35 time zone active
  • Green dots = most recent valid swing high (bull candle pivot)
  • Red dots = most recent valid swing low (bear candle pivot)
  • Signal fires when 23/29/35 bar creates a CHoCH at a local extreme
  • White dotted = 50% fib entry level (limit order here)
  • Light green dashed = TP1 (20pts) — move stop to breakeven when hit
  • Green dashed = TP2 (30pts) — runner target
  • Red dashed = Stop (10pts beyond extreme)
  • Label shows Entry / TP2 / SL prices
//@version=6
indicator("23/29/35 CHoCH Reversion | Newsie", overlay=true, max_bars_back=1000)

// INPUTS
stop_pts  = input.float(10.0, "Stop Loss (pts)",        group="Risk")
tp1_pts   = input.float(20.0, "TP1 / Move to BE (pts)", group="Risk")
tp2_pts   = input.float(30.0, "TP2 Runner (pts)",       group="Risk")
swing_len = input.int(5,      "Swing lookback bars",    group="Structure")
min_leg   = input.float(15.0, "Min leg size (pts)",     group="Structure")
line_len  = input.int(80,     "Line length (bars)",     group="Display")

// TIME ZONES (ET)
et_hour  = hour(time, "America/New_York")
et_min   = minute(time, "America/New_York")
h_plus_m = et_hour + et_min

in_23 = (h_plus_m >= 22 and h_plus_m <= 24) or (et_min >= 22 and et_min <= 24)
in_29 = (h_plus_m >= 28 and h_plus_m <= 30) or (et_min >= 28 and et_min <= 30)
in_35 = (h_plus_m >= 34 and h_plus_m <= 36) or (et_min >= 34 and et_min <= 36)
is_time  = in_23 or in_29 or in_35
zone_lbl = in_23 ? "23" : in_29 ? "29" : "35"

// CANDLE DIRECTION
bull_c = close > open
bear_c = close < open

// SWING PIVOTS (valid: bull candle high, bear candle low)
ph = ta.pivothigh(high, swing_len, swing_len)
pl = ta.pivotlow(low,  swing_len, swing_len)

bull_ph = not na(ph) and close[swing_len] > open[swing_len]
bear_pl = not na(pl)  and close[swing_len] < open[swing_len]

// Track most recent valid swing high/low
var float last_swing_high = na
var float last_swing_low  = na

if bull_ph
    last_swing_high := ph

if bear_pl
    last_swing_low := pl

// Draw swing levels (subtle dotted)
plot(not na(last_swing_high) ? last_swing_high : na, "Swing High",
     color=color.new(color.lime, 60), linewidth=1, style=plot.style_circles)
plot(not na(last_swing_low) ? last_swing_low : na, "Swing Low",
     color=color.new(color.red, 60), linewidth=1, style=plot.style_circles)

// CHoCH DETECTION at 23/29/35 bar
// SHORT: 23/29/35 bear candle at local high (above recent swing), leg from last swing low
// LONG:  23/29/35 bull candle at local low (below recent swing), leg from last swing high

lookback_high = ta.highest(high, swing_len)
lookback_low  = ta.lowest(low,   swing_len)

short_choch = is_time and bear_c and high >= lookback_high[1] and not na(last_swing_low)
long_choch  = is_time and bull_c and low  <= lookback_low[1]  and not na(last_swing_high)

// Leg size validation
short_leg = not na(last_swing_low)  ? high - last_swing_low  : na
long_leg  = not na(last_swing_high) ? last_swing_high - low  : na

short_valid = short_choch and not na(short_leg) and short_leg >= min_leg
long_valid  = long_choch  and not na(long_leg)  and long_leg  >= min_leg

// One signal per CHoCH (reset when new swing forms)
var bool short_taken = false
var bool long_taken  = false

if bull_ph: short_taken := false
if bear_pl: long_taken  := false

short_signal = short_valid and not short_taken
long_signal  = long_valid  and not long_taken

if short_signal: short_taken := true
if long_signal:  long_taken  := true

// Entry levels (50% fib of leg)
short_fib50 = short_signal ? high - short_leg * 0.5 : na
long_fib50  = long_signal  ? low  + long_leg  * 0.5 : na

// TP / Stop levels
short_stop = short_signal ? high + stop_pts : na
short_tp1  = short_signal ? short_fib50 - tp1_pts : na
short_tp2  = short_signal ? short_fib50 - tp2_pts : na

long_stop  = long_signal ? low - stop_pts : na
long_tp1   = long_signal ? long_fib50 + tp1_pts : na
long_tp2   = long_signal ? long_fib50 + tp2_pts : na

// Time bar highlight
bgcolor(is_time ? color.new(color.purple, 93) : na, title="Time Bar")

// Signal arrows
plotshape(long_signal,  "Long",  shape.triangleup,   location.belowbar,
          color=color.new(color.lime, 0), size=size.normal)
plotshape(short_signal, "Short", shape.triangledown,  location.abovebar,
          color=color.new(color.red,  0), size=size.normal)

// Draw lines on signal
var line fib_line  = na
var line tp1_line  = na
var line tp2_line  = na
var line sl_line   = na
atr = ta.atr(14)

if long_signal
    line.delete(fib_line)
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    fib_line := line.new(bar_index, long_fib50, bar_index+line_len, long_fib50,
                         color=color.new(color.white, 40), width=1, style=line.style_dotted)
    tp1_line := line.new(bar_index, long_tp1,   bar_index+line_len, long_tp1,
                         color=color.new(color.lime, 30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, long_tp2,   bar_index+line_len, long_tp2,
                         color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, long_stop,  bar_index+line_len, long_stop,
                         color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, low - atr*0.5,
              "LONG " + zone_lbl + "\nEntry: " + str.tostring(math.round(long_fib50,0)) +
              "\nTP2: "  + str.tostring(math.round(long_tp2,0)) +
              "\nSL: "   + str.tostring(math.round(long_stop,0)),
              style=label.style_label_up, color=color.new(color.lime,5),
              textcolor=color.black, size=size.tiny)

if short_signal
    line.delete(fib_line)
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    fib_line := line.new(bar_index, short_fib50, bar_index+line_len, short_fib50,
                         color=color.new(color.white, 40), width=1, style=line.style_dotted)
    tp1_line := line.new(bar_index, short_tp1,   bar_index+line_len, short_tp1,
                         color=color.new(color.lime, 30), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, short_tp2,   bar_index+line_len, short_tp2,
                         color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, short_stop,  bar_index+line_len, short_stop,
                         color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, high + atr*0.5,
              "SHORT " + zone_lbl + "\nEntry: " + str.tostring(math.round(short_fib50,0)) +
              "\nTP2: "  + str.tostring(math.round(short_tp2,0)) +
              "\nSL: "   + str.tostring(math.round(short_stop,0)),
              style=label.style_label_down, color=color.new(color.red,5),
              textcolor=color.white, size=size.tiny)

alertcondition(long_signal,  "23/29/35 LONG",  "CHoCH Time: LONG — Entry at 50% fib")
alertcondition(short_signal, "23/29/35 SHORT", "CHoCH Time: SHORT — Entry at 50% fib")
