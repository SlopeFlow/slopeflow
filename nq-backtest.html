NQ Scalper Pro | Backtest v2

Changes from v1:

  • EMA 200 trend filter - longs only above, shorts only below
  • Tighter RSI short filter - shorts need RSI above 55
  • Wider stops: 2.0x ATR stop, 3.0x ATR target (better R:R)
  • Cooldown increased to 8 bars
//@version=6
strategy("NQ Scalper Pro | Backtest v2", overlay=true, max_bars_back=500,
         initial_capital=50000, default_qty_type=strategy.fixed, default_qty_value=1,
         commission_type=strategy.commission.cash_per_contract, commission_value=2.05,
         slippage=2)

// INPUTS
session_filter  = input.bool(true, "Only trade RTH sessions", group="Session Filter")
session1_start  = input.int(9,  "Morning open hour ET",    group="Session Filter")
session1_end    = input.int(11, "Morning close hour ET",   group="Session Filter")
session2_start  = input.int(13, "Afternoon open hour ET",  group="Session Filter")
session2_end    = input.int(15, "Afternoon close hour ET", group="Session Filter")

ema_fast = input.int(9,  "Fast EMA", group="EMA Ribbon")
ema_slow = input.int(21, "Slow EMA", group="EMA Ribbon")
ema_trend = input.int(200, "Trend Filter EMA (HTF)", group="EMA Ribbon")

or_minutes = input.int(15, "Opening Range minutes", group="Opening Range", minval=5, maxval=60)

vol_ma_len = input.int(20,    "Volume MA Length",        group="Volume")
vol_mult   = input.float(1.5, "Volume Spike Multiplier", group="Volume")

atr_len    = input.int(14,    "ATR Length",        group="ATR Risk")
atr_stop   = input.float(2.0, "Stop Loss ATR x",   group="ATR Risk")
atr_target = input.float(3.0, "Take Profit ATR x", group="ATR Risk")

rsi_len      = input.int(14, "RSI Length",          group="RSI Filter")
rsi_ob       = input.int(65, "Overbought (tighter)", group="RSI Filter")
rsi_os       = input.int(35, "Oversold (tighter)",   group="RSI Filter")
rsi_short_ob = input.int(55, "Short only above RSI", group="RSI Filter")

cooldown_bars  = input.int(8, "Signal Cooldown Bars", group="Signal Filter", minval=1, maxval=20)
max_daily_loss = input.float(500.0, "Max Daily Loss $", group="Risk Management")

// SESSION
et_hour     = hour(time, "America/New_York")
et_min_full = hour(time, "America/New_York") * 60 + minute(time, "America/New_York")
in_morning   = et_hour >= session1_start and et_hour < session1_end
in_afternoon = et_hour >= session2_start and et_hour < session2_end
in_session   = not session_filter or (in_morning or in_afternoon)

new_day = ta.change(dayofmonth(time, "America/New_York")) != 0

// DAILY LOSS TRACKER
var float daily_start = strategy.equity
if new_day
    daily_start := strategy.equity
daily_pnl         = strategy.equity - daily_start
under_daily_limit = daily_pnl > -max_daily_loss

// EMA RIBBON
ema9   = ta.ema(close, ema_fast)
ema21  = ta.ema(close, ema_slow)
ema200 = ta.ema(close, ema_trend)

bull_ribbon  = ema9 > ema21
bear_ribbon  = ema9 < ema21
above_trend  = close > ema200
below_trend  = close < ema200

plot(ema9,   "EMA 9",   color=bull_ribbon ? color.new(color.aqua, 0) : color.new(color.red, 0), linewidth=1)
plot(ema21,  "EMA 21",  color=bull_ribbon ? color.new(color.teal, 30) : color.new(color.red, 30), linewidth=2)
plot(ema200, "EMA 200", color=color.new(color.white, 40), linewidth=2)

// VWAP
[vwap_val, _, _] = ta.vwap(hlc3, false, 1)
stdev = ta.stdev(close, 20)
plot(vwap_val,         "VWAP",    color=color.new(color.yellow, 0),  linewidth=2)
plot(vwap_val + stdev, "VWAP+1s", color=color.new(color.green, 50),  linewidth=1)
plot(vwap_val - stdev, "VWAP-1s", color=color.new(color.red,   50),  linewidth=1)

// OPENING RANGE
rth_open_min = 9 * 60 + 30
or_close_min = rth_open_min + or_minutes
is_rth_open  = et_min_full == rth_open_min
in_or_window = et_min_full >= rth_open_min and et_min_full < or_close_min
or_just_done = et_min_full == or_close_min

var float or_high = na
var float or_low  = na
var bool  or_set  = false

if new_day
    or_high := na
    or_low  := na
    or_set  := false

if is_rth_open
    or_high := high
    or_low  := low
    or_set  := false
else if in_or_window and not or_set
    or_high := math.max(or_high, high)
    or_low  := math.min(or_low, low)
else if or_just_done and not or_set
    or_set  := true

plot(or_set ? or_high : na, "OR High", color=color.new(color.orange, 0), linewidth=2, style=plot.style_stepline)
plot(or_set ? or_low  : na, "OR Low",  color=color.new(color.orange, 0), linewidth=2, style=plot.style_stepline)

// VOLUME + RSI + ATR
vol_ma    = ta.sma(volume, vol_ma_len)
vol_spike = volume > vol_ma * vol_mult
rsi          = ta.rsi(close, rsi_len)
rsi_ok_long  = rsi < rsi_ob
rsi_ok_short = rsi > rsi_os and rsi > rsi_short_ob
atr          = ta.atr(atr_len)

// PRE-COMPUTE CROSSOVERS
cross_vwap_up = ta.crossover(close, vwap_val)
cross_vwap_dn = ta.crossunder(close, vwap_val)
cross_or_high = ta.crossover(close, or_high)
cross_or_low  = ta.crossunder(close, or_low)

// SIGNALS - longs only when above EMA200, shorts only when below EMA200
vwap_reclaim_long  = cross_vwap_up and bull_ribbon and vol_spike and rsi_ok_long  and in_session and above_trend
vwap_reject_short  = cross_vwap_dn and bear_ribbon and vol_spike and rsi_ok_short and in_session and below_trend
or_breakout_long   = or_set and cross_or_high and bull_ribbon and vol_spike and in_session and above_trend
or_breakout_short  = or_set and cross_or_low  and bear_ribbon and vol_spike and in_session and below_trend
ema_pullback_long  = bull_ribbon and low <= ema9 and close > ema9 and close > close[1] and vol_spike and rsi_ok_long  and in_session and above_trend
ema_pullback_short = bear_ribbon and high >= ema9 and close < ema9 and close < close[1] and vol_spike and rsi_ok_short and in_session and below_trend

raw_long  = vwap_reclaim_long or or_breakout_long or ema_pullback_long
raw_short = vwap_reject_short or or_breakout_short or ema_pullback_short

// COOLDOWN
var int last_signal_bar = -999
on_cooldown  = (bar_index - last_signal_bar) < cooldown_bars
long_signal  = raw_long  and not on_cooldown and under_daily_limit
short_signal = raw_short and not on_cooldown and under_daily_limit

if long_signal or short_signal
    last_signal_bar := bar_index

// ENTRIES & EXITS
if long_signal and strategy.position_size == 0
    strategy.entry("Long", strategy.long)
    strategy.exit("Long Exit", "Long", stop=close - atr * atr_stop, limit=close + atr * atr_target)

if short_signal and strategy.position_size == 0
    strategy.entry("Short", strategy.short)
    strategy.exit("Short Exit", "Short", stop=close + atr * atr_stop, limit=close - atr * atr_target)

// CLOSE at session end
if et_hour >= 15 and strategy.position_size != 0
    strategy.close_all("Session End")
