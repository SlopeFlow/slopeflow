Zone Reversal System v2 | Newsie

  • White line = 50% zone midpoint
  • Green stepline = leg high | Red stepline = leg low
  • Orange X = invalid high/low (wrong candle color)
  • Purple bar = 23/29/35 time zone active
  • Signal = time bar touches 50% zone
  • Stop 10pts | TP1 20pts (move to BE) | TP2 30pts
//@version=6
indicator("23/29/35 Zone System | Newsie", overlay=true, max_bars_back=1000)

// INPUTS
stop_pts  = input.float(10.0, "Stop Loss (pts)",      group="Risk")
tp1_pts   = input.float(20.0, "TP1 BE level (pts)",   group="Risk")
tp2_pts   = input.float(30.0, "TP2 Runner (pts)",     group="Risk")
line_len  = input.int(200,    "Zone line length",      group="Display")

// ─── TIME CALCULATION ─────────────────────────────────────────
et_h = hour(time,   "America/New_York")
et_m = minute(time, "America/New_York")

h_plus_m = et_h + et_m
m_minus_h = et_m - et_h
min_only  = et_m

// Drop-zero method (specific times only)
// 02:02-04 = 22-24, 02:08-09 = 28-29, 03:04-06 = 34-36
drop_zero = ((et_h == 2 and et_m >= 2 and et_m <= 4) or (et_h == 2 and et_m >= 8 and et_m <= 9) or (et_h == 3 and et_m >= 4 and et_m <= 6))

drop_zero_val = et_h * 10 + et_m  // e.g. 2*10+3 = 23

in_23_dz = drop_zero and (drop_zero_val >= 22 and drop_zero_val <= 24)
in_29_dz = drop_zero and (drop_zero_val >= 28 and drop_zero_val <= 29)
in_35_dz = drop_zero and (drop_zero_val >= 34 and drop_zero_val <= 36)

in_23 = ((h_plus_m >= 22 and h_plus_m <= 24) or (m_minus_h >= 22 and m_minus_h <= 24) or (min_only >= 22 and min_only <= 24) or in_23_dz)
in_29 = ((h_plus_m >= 28 and h_plus_m <= 30) or (m_minus_h >= 28 and m_minus_h <= 30) or (min_only >= 28 and min_only <= 30) or in_29_dz)
in_35 = ((h_plus_m >= 34 and h_plus_m <= 36) or (m_minus_h >= 34 and m_minus_h <= 36) or (min_only >= 34 and min_only <= 36) or in_35_dz)

is_time  = in_23 or in_29 or in_35
zone_lbl = in_23 ? "23" : in_29 ? "29" : "35"

// ─── CANDLE DIRECTION ─────────────────────────────────────────
bull_c = close > open
bear_c = close < open

// ─── LEG TRACKING ─────────────────────────────────────────────
// Track the current leg: one anchor high (bull candle) + one anchor low (bear candle)
// Update high on each new HH with bull candle (keep low anchor)
// Update low on each new LL with bear candle (keep high anchor)
// Reset both when 50% is touched (zone used) or opposite CHoCH

var float leg_high     = na   // current leg high (bull candle)
var float leg_low      = na   // current leg low (bear candle)
var float zone_50      = na   // 50% midpoint of leg
var bool  long_active  = false // tracking upleg (longs)
var bool  short_active = false // tracking downleg (shorts)
var bool  zone_touched = false // 50% has been touched this zone

new_day = ta.change(dayofmonth(time, "America/New_York")) != 0

if new_day
    leg_high     := na
    leg_low      := na
    zone_50      := na
    long_active  := false
    short_active := false
    zone_touched := false

// New valid high: bull candle that is higher than recent bars
is_new_hh = bull_c and high > ta.highest(high[1], 5)

// New valid low: bear candle that is lower than recent bars
is_new_ll = bear_c and low < ta.lowest(low[1], 5)

// UPLEG: new HH with bull candle
if is_new_hh
    if not long_active
        // Start fresh upleg — anchor low = most recent bear candle low
        leg_low    := ta.lowest(low, 10)
        long_active  := true
        short_active := false
        zone_touched := false
    // Update high anchor (keep same low)
    leg_high := high
    zone_50  := (leg_high + leg_low) / 2

// DOWNLEG: new LL with bear candle
if is_new_ll
    if not short_active
        // Start fresh downleg — anchor high = most recent bull candle high
        leg_high     := ta.highest(high, 10)
        short_active := true
        long_active  := false
        zone_touched := false
    // Update low anchor (keep same high)
    leg_low := low
    zone_50 := (leg_high + leg_low) / 2

// ─── ZONE DISPLAY ─────────────────────────────────────────────
// Draw 50% zone as a line
plot(not na(zone_50) ? zone_50 : na, "50% Zone",
     color=color.new(color.white, 40), linewidth=1, style=plot.style_line)

// Draw leg high/low
plot(not na(leg_high) ? leg_high : na, "Leg High",
     color=color.new(color.lime, 50), linewidth=1, style=plot.style_stepline)
plot(not na(leg_low) ? leg_low : na, "Leg Low",
     color=color.new(color.red, 50), linewidth=1, style=plot.style_stepline)

// ─── SIGNAL: 23/29/35 bar touches 50% zone ─────────────────────
zone_tolerance = 5.0  // within 5pts of 50% counts as "touching"

at_50_long  = long_active  and not na(zone_50) and low  <= zone_50 + zone_tolerance and low  >= zone_50 - zone_tolerance
at_50_short = short_active and not na(zone_50) and high >= zone_50 - zone_tolerance and high <= zone_50 + zone_tolerance

long_signal  = is_time and at_50_long  and bull_c
short_signal = is_time and at_50_short and bear_c

// Mark zone touched
if at_50_long or at_50_short
    zone_touched := true

// Time bar highlight
bgcolor(is_time ? color.new(color.purple, 93) : na, title="Time Bar")

// Invalid high/low markers (wrong candle color)
invalid_high = is_new_hh and bear_c  // high made by bear candle = invalid
invalid_low  = is_new_ll and bull_c  // low made by bull candle = invalid

plotshape(invalid_high, "Invalid High", shape.xcross, location.abovebar,
          color=color.new(color.orange, 20), size=size.tiny)
plotshape(invalid_low,  "Invalid Low",  shape.xcross, location.belowbar,
          color=color.new(color.orange, 20), size=size.tiny)

// Signal arrows
plotshape(long_signal,  "Long",  shape.triangleup,   location.belowbar,
          color=color.new(color.lime, 0), size=size.normal)
plotshape(short_signal, "Short", shape.triangledown,  location.abovebar,
          color=color.new(color.red,  0), size=size.normal)

// TP / Stop lines
var line tp1_line = na
var line tp2_line = na
var line sl_line  = na
atr = ta.atr(14)

if long_signal
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, close + tp1_pts, bar_index + line_len, close + tp1_pts,
                         color=color.new(color.lime, 40), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, close + tp2_pts, bar_index + line_len, close + tp2_pts,
                         color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, close - stop_pts, bar_index + line_len, close - stop_pts,
                         color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, low - atr * 0.5,
              "LONG " + zone_lbl + "
SL:" + str.tostring(math.round(close - stop_pts, 0)) +
              "  TP2:" + str.tostring(math.round(close + tp2_pts, 0)),
              style=label.style_label_up, color=color.new(color.lime, 5),
              textcolor=color.black, size=size.tiny)

if short_signal
    line.delete(tp1_line)
    line.delete(tp2_line)
    line.delete(sl_line)
    tp1_line := line.new(bar_index, close - tp1_pts, bar_index + line_len, close - tp1_pts,
                         color=color.new(color.lime, 40), width=1, style=line.style_dashed)
    tp2_line := line.new(bar_index, close - tp2_pts, bar_index + line_len, close - tp2_pts,
                         color=color.new(color.lime, 0),  width=2, style=line.style_dashed)
    sl_line  := line.new(bar_index, close + stop_pts, bar_index + line_len, close + stop_pts,
                         color=color.new(color.red,  20), width=1, style=line.style_dashed)
    label.new(bar_index, high + atr * 0.5,
              "SHORT " + zone_lbl + "
SL:" + str.tostring(math.round(close + stop_pts, 0)) +
              "  TP2:" + str.tostring(math.round(close - tp2_pts, 0)),
              style=label.style_label_down, color=color.new(color.red, 5),
              textcolor=color.white, size=size.tiny)

alertcondition(long_signal,  "Zone LONG",  "23/29/35 Zone: LONG signal")
alertcondition(short_signal, "Zone SHORT", "23/29/35 Zone: SHORT signal")
